144 template <typename Callback, bool Parallel>
145 auto tree_enumerator<Callback, Parallel>::iterate(bipartition_iterator& bip_it,
146  const bitvector& new_constraint_occ)
147  -> result_type {
148  if (m_cb.fast_return(bip_it)) {
149  return m_cb.fast_return_value(bip_it);
150  }
151 
152  auto result = m_cb.begin_iteration(bip_it, new_constraint_occ, *m_constraints);
153  if (should_resume_parallel(bip_it)) {
154  // initialize work queue
155  assert(m_work_pool.empty());

157  m_work_pool.emplace_back(bip_it.get_current_set(), m_cb.null_result());
158  bip_it.flip_sets();
159  m_work_pool.emplace_back(bip_it.get_current_set(), m_cb.null_result());
160  bip_it.increase();
161  }
162  m_next_work.store(0);
163 
164  // initialize threads
165  const index num_threads =
166  std::min<index>(m_work_pool.size(), std::thread::hardware_concurrency());
167  const index max_work = m_work_pool.size();
168  for (index i = 0; i < num_threads; ++i) {
169  m_thread_pool.emplace_back([&, i]() {
170  index work_idx;
171  auto& runner = m_runner_pool[i];
172  while ((work_idx = m_next_work.fetch_add(1)) < max_work) {
173  auto& work = m_work_pool[work_idx];

175  }
176  });
177  }
178  for (auto& thread : m_thread_pool) {
179  thread.join();
180  }
181  for (index i = 0; i < max_work; i += 2) {
182  auto& left = m_work_pool[i].second;
183  auto& right = m_work_pool[i + 1].second;
184  result = m_cb.accumulate(result, m_cb.combine(left, right));
185  }
186  m_thread_pool.clear();
187  m_work_pool.clear();
188  } else {
189  // iterate over all possible bipartitions

191  m_cb.step_iteration(bip_it);
192  m_cb.left_subcall();
193  auto left = run(bip_it.get_current_set(), new_constraint_occ);
194 
195  bip_it.flip_sets();
196  m_cb.right_subcall();
197  auto right = run(bip_it.get_current_set(), new_constraint_occ);
198 
199  // accumulate result
200  result = m_cb.accumulate(result, m_cb.combine(left, right));
201 
202  bip_it.increase();
203  }
204  }
205  m_cb.finish_iteration();
