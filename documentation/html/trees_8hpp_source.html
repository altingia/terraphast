156 void foreach_postorder(const tree& t, F cb) {
157  index root_idx = 0;
158  assert(is_root(t[root_idx]));
159 
160  enum class visited { none, left, both };
161 
162  std::stack<std::pair<index, visited>> stack;
163  stack.push(std::make_pair(root_idx, visited::none));
164  while (!stack.empty()) {
165  auto current = stack.top();
166  auto cur_idx = current.first;
167  stack.pop();
168  auto cur_node = t[cur_idx];
169  if (is_leaf(cur_node)) {
170  cb(current.first);
171  } else {
172  switch (current.second) {
173  // descend into left tree
174  case visited::none:
175  stack.push(std::make_pair(cur_idx, visited::left));
176  stack.push(std::make_pair(cur_node.lchild(), visited::none));
177  break;
178  // descend into right tree
179  case visited::left:
180  stack.push(std::make_pair(cur_idx, visited::both));
181  stack.push(std::make_pair(cur_node.rchild(), visited::none));
182  break;
183  // move up again
184  case visited::both:
185  cb(cur_idx);
186  break;
187  }
188  }
189  }
